\section{Trigonometria}
\begin{gather*}
 \tan (\alpha) = \frac{\sin \alpha}{\cos \alpha} \\
 \cot (\alpha) = \frac{\cos \alpha}{\sin \alpha} \\
 \sin (\alpha) \cos (\alpha) = \sin (2 \alpha) \\
 \textbf{Area cerchio: } A = r^2 \cdot \pi \\
 \textbf{Volume sfera: } V = \frac{4}{3} \pi r^3
\end{gather*}
